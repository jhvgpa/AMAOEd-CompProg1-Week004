#include <iostream>

using namespace std;

int main()
{
   int user_num;
   
   cout << "Enter a number from 10-20: ";
   cin >> user_num;
   while (user_num >= 0) {
       cout << "> " << user_num << endl;
       user_num = user_num - 1;
   }
   
   return 0;
}
#include <iostream>

using namespace std;

int main()
{
   int user_num;
   
   cout << "Enter a number from 10-20: ";
   cin >> user_num;
   cout << user_num << endl;
   for (int i = 0; i <= user_num; i = i + 1  ) {
       cout << "> " << i << endl;
   }
   
   return 0;
}

{
   int user_num;
   
   while (true) {
       cout << "Enter a number from 10-20: ";
       cin >> user_num;
       if (user_num > 20 || user_num < 10) {
           continue;
       }
       else {
           return user_num;
           break;
       }
   cout << user_num << endl;
   for (int i = 0; i <= user_num; i = i + 1  ) {
       cout << "> " << i << endl;
   }
   }
   return 0;
}
